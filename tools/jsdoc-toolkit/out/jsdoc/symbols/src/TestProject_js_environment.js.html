 20 function rand(maxNum){
 21      return Math.round(maxNum*Math.random());
 22 }
 23 
 24 function createNewEnvironment(){
 25 
 26      size = rand(maxSize);
 27      creatureList[0] = new Creature();
 28      creatureList[0].createAt("thisId1", 0, rand(maxWidth-size),
 29                 rand(maxHeight-size), size, 
 30                 rand(255), rand(255), rand(255),
 31                 -10 + rand(20), -10 + rand(20));
 32 
 33      size = rand(maxSize);
 34      creatureList[1] = new Creature();
 35      creatureList[1].createAt("thisId2", 1, rand(maxWidth-size),
 36                 rand(maxHeight-size), size, 
 37                 rand(255), rand(255), rand(255),
 38                 -10 + rand(20), -10 + rand(20));
 39 
 40      setTimeout("nextTurn()", interval);
 41 }
 42 
 43 function nextTurn(){
 44 
 45      for(var thisCreatureIndex =0; 
 46          thisCreatureIndex < creatureList.length; 
 47          thisCreatureIndex++) {
 48           
 49          creatureList[thisCreatureIndex].move();
 50      }
 51      
 52      if (ison){
 53          setTimeout("nextTurn()", interval);
 54      }
 55      elapsedTurns++;
 56 }
