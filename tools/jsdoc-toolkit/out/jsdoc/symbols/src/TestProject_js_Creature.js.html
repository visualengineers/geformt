1 function Creature(){
  2 
  3    this.x = 30;
  4    this.y = 50;
  5    this.colorRed=0;
  6    this.colorGreen=0;
  7    this.colorBlue=0;
  8    this.xspeed = 2;
  9    this.yspeed = 3;
 10    this.size = 20;
 11    this.status = 1;
 12    this.maxage = 500;
 13    this.currentAge = 0;
 14    this.id = "thisId";
 15    
 16    this.bouncex = function(){
 17        this.xspeed = -1 * this.xspeed;
 18    };
 19    this.bouncey = function(){
 20        this.yspeed = -1 * this.yspeed;
 21    };
 22    
 23    this.die = function(){
 24        this.status = 0;
 25        grid[this.x][this.y] = -1;
 26            
 27        this.xspeed = 0;
 28        this.yspeed = 0;
 29        this.size = 0;
 30        theDiv = document.getElementById(this.id);
 31        theDiv.style.backgroundColor = "white";
 32        theDiv.style.border = "0px";
 33    };
 34    
 35    this.move = function (){
 36         
 37        this.currentAge++;
 38        if (this.currentAge > this.maxage){
 39           this.die();
 40        } else {
 41         
 42            this.x = this.x + this.xspeed;
 43            this.y = this.y + this.yspeed;
 44              
 45            if ((this.x + this.size) > maxWidth ){
 46                this.x = maxWidth - this.size;
 47                this.bouncex();
 48            } else if (this.x < 0) {
 49                this.x = 0;
 50                this.bouncex();
 51            } else if ((this.y + this.size) > maxHeight ){
 52                this.y = maxHeight - this.size;
 53                this.bouncey();             
 54            } else if (this.y < 0) {
 55                this.y = 0;
 56                this.bouncey();
 57            } 
 58              
 59            this.renderAt(this.x, this.y);
 60            
 61            this.checkForPartner();
 62        }
 63    };
 64 
 65    this.renderAt = function(xpos, ypos){
 66        theDiv = document.getElementById(this.id);
 67        theDiv.style.left = this.x;
 68        theDiv.style.top = this.y;
 69    };  
 70

116 this.breedWith = function(otherCreature){
117         
118 		if (elapsedTurns < 50){
119            // do nothing

123            babyxspeed = this.getNewValue(this.xspeed,

131            babycolorgreen = this.getNewValue(this.colorGreen,

147    this.checkForPartner = function(){
148         
149        xmax = this.x+size;
150        if (this.x+size > maxWidth){
151            xmax = maxWidth;
152        } 
153         
154        ymax = this.y+size;
155        if (this.y+size > maxHeight){
156           ymax = maxHeight;
157        } 
158         
159        for(var xtocheck = this.x; xtocheck < xmax; xtocheck++) {
160           for(var ytocheck = this.y; ytocheck < ymax; ytocheck++) {
161               if (grid[xtocheck][ytocheck] != -1 &&
162                   grid[xtocheck][ytocheck] != this.gridIndex){
163 
164                   this.breedWith(
165                         creatureList[grid[xtocheck][ytocheck]]);
166                   break;
167               }
168           }                
169       }
170    };

172 this.getNewValue = function(old1, old2, min, max){
173        if (old1 < old2){
174           low = old1;
175           high = old2;
176        } else {
177           low = old2;
178           high = old1;
179        }
180           
181        low = low - Math.round(low * .15);
182        high = high + Math.round(high * .15);
183           
184        if (low < min){
185           low = min;
186        }
187        if (high > max){
188           high = max;
189        }
190           
191        spread = high - low;
192        offset = Math.round(Math.random()*spread);
193        return low + offset;
194    };
